🧩 II. Lỗi 1 — CSP chặn Cloudflare beacon

Log:

Loading the script 'https://static.cloudflareinsights.com/beacon.min.js' violates CSP.

🔍 Nguyên nhân

CSP (Content Security Policy) của bạn không cho phép load script ngoài domain self, trong khi Cloudflare analytics dùng domain khác.

✅ Cách xử lý

Nếu bạn KHÔNG cần Cloudflare tracking:
➡️ Xóa dòng sau khỏi _document.js hoặc <head>:

<script defer src="https://static.cloudflareinsights.com/beacon.min.js"></script>


Nếu bạn CẦN giữ lại tracking:
➡️ Thêm domain vào CSP (Next.js next.config.js):

async headers() {
  return [
    {
      source: "/(.*)",
      headers: [
        {
          key: "Content-Security-Policy",
          value:
            "script-src 'self' 'unsafe-inline' 'unsafe-eval' https://static.cloudflareinsights.com;",
        },
      ],
    },
  ];
}

🧩 III. Lỗi 2 — Upload ảnh thất bại (400 Bad Request)

Log:

POST https://tapchinckhhcqs.abacusai.app/api/media 400 (Bad Request)
Upload failed

🔍 Nguyên nhân

RẤT có thể do API backend không đọc được FormData, hoặc file không có trong body.

Đây là nguyên nhân thực khiến bạn “upload ảnh không được”.

✅ Giải pháp A — Kiểm tra client (frontend)

Trong component upload (React/Next.js):

async function handleUpload(e) {
  const file = e.target.files[0];
  if (!file) return alert("Chưa chọn ảnh");

  const formData = new FormData();
  formData.append("file", file);

  try {
    const res = await fetch("/api/media", {
      method: "POST",
      body: formData,
    });

    if (!res.ok) {
      throw new Error("Upload failed");
    }

    const data = await res.json();
    console.log("✅ Upload thành công:", data);
  } catch (err) {
    console.error("❌ Upload lỗi:", err);
  }
}

✅ Giải pháp B — Kiểm tra server (Next.js API Route /api/media)

Tạo file pages/api/media.js:

import formidable from "formidable";
import fs from "fs";
import path from "path";

export const config = {
  api: { bodyParser: false },
};

export default async function handler(req, res) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  const uploadDir = path.join(process.cwd(), "public/uploads");

  fs.mkdirSync(uploadDir, { recursive: true });

  const form = new formidable.IncomingForm({ uploadDir, keepExtensions: true });

  form.parse(req, (err, fields, files) => {
    if (err || !files.file) {
      console.error("❌ Upload thất bại:", err);
      return res.status(400).json({ error: "Upload failed" });
    }

    const filePath = path.basename(files.file[0].filepath);
    res.status(200).json({ url: `/uploads/${filePath}` });
  });
}


⚠️ Quan trọng: phải có export const config = { api: { bodyParser: false } }
Nếu thiếu, Next.js sẽ không đọc được file binary từ FormData → gây lỗi 400.

import formidable from "formidable";
import fs from "fs";
import path from "path";

// ⛔ Phải tắt bodyParser để xử lý FormData thủ công
export const config = {
  api: { bodyParser: false },
};

export default async function handler(req, res) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  try {
    const uploadDir = path.join(process.cwd(), "public/uploads");
    if (!fs.existsSync(uploadDir)) fs.mkdirSync(uploadDir, { recursive: true });

    const form = new formidable.IncomingForm({
      uploadDir,
      keepExtensions: true,
      maxFileSize: 5 * 1024 * 1024, // 5MB
    });

    form.parse(req, (err, fields, files) => {
      if (err || !files.file) {
        console.error("❌ Upload lỗi:", err);
        return res.status(400).json({ error: "Upload failed" });
      }

      const fileData = Array.isArray(files.file) ? files.file[0] : files.file;
      const fileName = path.basename(fileData.filepath);
      const fileUrl = `/uploads/${fileName}`;

      console.log("✅ Upload thành công:", fileUrl);
      res.status(200).json({ url: fileUrl });
    });
  } catch (e) {
    console.error("🚨 Lỗi hệ thống upload:", e);
    res.status(500).json({ error: "Internal Server Error" });
  }
}

import { EditorContent, useEditor } from "@tiptap/react";
import StarterKit from "@tiptap/starter-kit";
import Link from "@tiptap/extension-link";

export default function ArticleEditor({ value, onChange }) {
  const editor = useEditor({
    extensions: [
      StarterKit.configure({
        link: false, // Tránh trùng extension
      }),
      Link.configure({
        openOnClick: false,
      }),
    ],
    content: value || "",
    onUpdate: ({ editor }) => {
      const html = editor.getHTML();
      if (onChange) onChange(html);
    },
  });

  // 🚫 Fix React error #418: kiểm tra editor trước khi render
  if (!editor) return null;

  return (
    <div className="border border-gray-300 rounded-md p-3 bg-white">
      <EditorContent editor={editor} />
    </div>
  );
}

import { useState } from "react";

export default function ImageUpload({ onUploaded }) {
  const [uploading, setUploading] = useState(false);

  const handleUpload = async (e) => {
    const file = e.target.files[0];
    if (!file) return alert("Vui lòng chọn ảnh!");

    const formData = new FormData();
    formData.append("file", file);
    setUploading(true);

    try {
      const res = await fetch("/api/media", {
        method: "POST",
        body: formData,
      });

      if (!res.ok) throw new Error(`Upload failed: ${res.status}`);

      const data = await res.json();
      console.log("✅ File uploaded:", data);
      onUploaded?.(data.url);
    } catch (err) {
      console.error("❌ Upload lỗi:", err);
      alert("Tải ảnh thất bại, vui lòng thử lại!");
    } finally {
      setUploading(false);
    }
  };

  return (
    <div className="flex flex-col items-start gap-2">
      <label className="text-sm font-semibold">Tải ảnh lên:</label>
      <input
        type="file"
        accept="image/*"
        onChange={handleUpload}
        disabled={uploading}
      />
      {uploading && <p className="text-gray-500 text-xs">Đang tải ảnh...</p>}
    </div>
  );
}
