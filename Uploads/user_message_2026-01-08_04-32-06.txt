📘 A. Cấu trúc bảng mới trong Prisma
model PlagiarismReport {
  id             String   @id @default(cuid())
  submission     Submission @relation(fields: [submissionId], references: [id])
  submissionId   String
  checkedBy      String?   // userId của người yêu cầu
  similarityRate Float?
  reportUrl      String?
  createdAt      DateTime  @default(now())
  updatedAt      DateTime  @updatedAt
  status         String    @default("PENDING") // PENDING | COMPLETED | FAILED
}

🧱 B. API Route: /app/api/plagiarism/check/route.ts
import { prisma } from "@/lib/prisma"
import { NextResponse } from "next/server"
import { performPlagiarismCheck } from "@/lib/plagiarism/turnitin"

export async function POST(req: Request) {
  const { submissionId, fileUrl, userId } = await req.json()

  // Lưu yêu cầu ban đầu
  const report = await prisma.plagiarismReport.create({
    data: { submissionId, checkedBy: userId }
  })

  // Gửi file đến dịch vụ đạo văn (ví dụ Turnitin API hoặc Grammarly)
  const result = await performPlagiarismCheck(fileUrl)

  await prisma.plagiarismReport.update({
    where: { id: report.id },
    data: {
      similarityRate: result.similarityRate,
      reportUrl: result.reportUrl,
      status: "COMPLETED"
    }
  })

  return NextResponse.json({ success: true, result })
}

C. 3️⃣ Viết hàm so sánh nội dung giữa các bài

Tạo /lib/plagiarism/internalCheck.ts:

import { prisma } from "@/lib/prisma"

// Simple cosine similarity (TF-IDF style)
function cosineSimilarity(a: string, b: string): number {
  const wordsA = a.toLowerCase().split(/\W+/)
  const wordsB = b.toLowerCase().split(/\W+/)
  const allWords = new Set([...wordsA, ...wordsB])

  const vecA = Array.from(allWords).map(w => wordsA.filter(x => x === w).length)
  const vecB = Array.from(allWords).map(w => wordsB.filter(x => x === w).length)

  const dot = vecA.reduce((sum, v, i) => sum + v * vecB[i], 0)
  const magA = Math.sqrt(vecA.reduce((s, v) => s + v * v, 0))
  const magB = Math.sqrt(vecB.reduce((s, v) => s + v * v, 0))

  return magA && magB ? dot / (magA * magB) : 0
}

export async function checkInternalPlagiarism(submissionId: string) {
  const submission = await prisma.submission.findUnique({
    where: { id: submissionId },
    include: { article: true }
  })
  if (!submission) throw new Error("Submission not found")

  const textA = submission.article?.htmlBody || submission.abstractVn || submission.title
  if (!textA) return { score: 0, matches: [] }

  const others = await prisma.submission.findMany({
    where: { id: { not: submissionId } },
    include: { article: true }
  })

  const matches = others.map(s => {
    const textB = s.article?.htmlBody || s.abstractVn || s.title
    if (!textB) return null
    const sim = cosineSimilarity(textA, textB)
    return sim > 0.2
      ? { id: s.id, title: s.title, similarity: Math.round(sim * 100) }
      : null
  }).filter(Boolean)

  const maxScore = matches.reduce((max, m) => Math.max(max, m!.similarity), 0)
  return { score: maxScore, matches }
}

D. 4️⃣ API endpoint kiểm tra đạo văn nội bộ

/app/api/plagiarism/check/route.ts:

import { NextResponse } from "next/server"
import { checkInternalPlagiarism } from "@/lib/plagiarism/internalCheck"
import { prisma } from "@/lib/prisma"

export async function POST(req: Request) {
  const { submissionId, userId } = await req.json()

  const result = await checkInternalPlagiarism(submissionId)

  await prisma.plagiarismReport.create({
    data: {
      submissionId,
      checkedBy: userId,
      similarityScore: result.score,
      details: result.matches,
      status: "COMPLETED"
    }
  })

  return NextResponse.json(result)
}

Ví dụ /lib/plagiarism/copyleaks.ts:

import axios from "axios"

export async function checkExternalPlagiarism(fileUrl: string) {
  const token = await getCopyleaksToken()
  const resp = await axios.post(
    "https://api.copyleaks.com/v3/scans/submit/url",
    { url: fileUrl },
    { headers: { Authorization: `Bearer ${token}` } }
  )
  return resp.data
}
